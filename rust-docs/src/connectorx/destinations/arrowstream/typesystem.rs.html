1use crate::impl_typesystem;
2use chrono::{DateTime, NaiveDate, NaiveDateTime, NaiveTime, Utc};
3use rust_decimal::Decimal;
4
5#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
6pub enum ArrowTypeSystem {
7    Int32(bool),
8    Int64(bool),
9    UInt32(bool),
10    UInt64(bool),
11    Float32(bool),
12    Float64(bool),
13    Decimal(bool),
14    Boolean(bool),
15    LargeUtf8(bool),
16    LargeBinary(bool),
17    Date32(bool),
18    Date64(bool),
19    Time64(bool),
20    DateTimeTz(bool),
21    Float32Array(bool),
22}
23
24impl_typesystem! {
25    system = ArrowTypeSystem,
26    mappings = {
27        { Int32           => i32                }
28        { Int64           => i64                }
29        { UInt32          => u32                }
30        { UInt64          => u64                }
31        { Float64         => f64                }
32        { Float32         => f32                }
33        { Decimal         => Decimal            }
34        { Boolean         => bool               }
35        { LargeUtf8       => String             }
36        { LargeBinary     => Vec<u8>            }
37        { Date32          => NaiveDate          }
38        { Date64          => NaiveDateTime      }
39        { Time64          => NaiveTime          }
40        { DateTimeTz      => DateTime<Utc>      }
41        { Float32Array    => Vec<Option<f32>>   }
42    }
43}
