37impl_typesystem! {
38    system = MySQLTypeSystem,
39    mappings = {
40        { Tiny => i8 }
41        { Short | Year => i16 }
42        { Long | Int24 => i32}
43        { LongLong => i64 }
44        { Float => f32 }
45        { Double => f64 }
46        { UTiny => u8 }
47        { UShort => u16 }
48        { ULong | UInt24 => u32}
49        { ULongLong => u64 }
50        { Datetime | Timestamp => NaiveDateTime }
51        { Date => NaiveDate }
52        { Time => NaiveTime }
53        { Decimal => Decimal }
54        { Char | VarChar | Enum => String }
55        { TinyBlob | Blob | MediumBlob | LongBlob | Bit => Vec<u8>}
56        { Json => Value }
57    }
58}

60impl<'a> From<(&'a ColumnType, &'a ColumnFlags)> for MySQLTypeSystem {
61    fn from(col: (&'a ColumnType, &'a ColumnFlags)) -> MySQLTypeSystem {
62        use MySQLTypeSystem::*;
63        let (ty, flag) = col;
64        let null_ok = !flag.contains(ColumnFlags::NOT_NULL_FLAG);
65        let unsigned = flag.contains(ColumnFlags::UNSIGNED_FLAG);
66        match ty {
67            ColumnType::MYSQL_TYPE_TINY => {
68                if unsigned {
69                    UTiny(null_ok)
70                } else {
71                    Tiny(null_ok)
72                }
73            }
74            ColumnType::MYSQL_TYPE_SHORT => {
75                if unsigned {
76                    UShort(null_ok)
77                } else {
78                    Short(null_ok)
79                }
80            }
81            ColumnType::MYSQL_TYPE_INT24 => {
82                if unsigned {
83                    UInt24(null_ok)
84                } else {
85                    Int24(null_ok)
86                }
87            }
88            ColumnType::MYSQL_TYPE_LONG => {
89                if unsigned {
90                    ULong(null_ok)
91                } else {
92                    Long(null_ok)
93                }
94            }
95            ColumnType::MYSQL_TYPE_LONGLONG => {
96                if unsigned {
97                    ULongLong(null_ok)
98                } else {
99                    LongLong(null_ok)
100                }
101            }
102            ColumnType::MYSQL_TYPE_FLOAT => Float(null_ok),
103            ColumnType::MYSQL_TYPE_DOUBLE => Double(null_ok),
104            ColumnType::MYSQL_TYPE_DATETIME => Datetime(null_ok),
105            ColumnType::MYSQL_TYPE_DATE => Date(null_ok),
106            ColumnType::MYSQL_TYPE_TIME => Time(null_ok),
107            ColumnType::MYSQL_TYPE_DECIMAL => Decimal(null_ok),
108            ColumnType::MYSQL_TYPE_NEWDECIMAL => Decimal(null_ok),
109            ColumnType::MYSQL_TYPE_STRING => Char(null_ok),
110            ColumnType::MYSQL_TYPE_VAR_STRING => VarChar(null_ok),
111            ColumnType::MYSQL_TYPE_TIMESTAMP => Timestamp(null_ok),
112            ColumnType::MYSQL_TYPE_YEAR => Year(null_ok),
113            ColumnType::MYSQL_TYPE_ENUM => Enum(null_ok),
114            ColumnType::MYSQL_TYPE_TINY_BLOB => TinyBlob(null_ok),
115            ColumnType::MYSQL_TYPE_BLOB => Blob(null_ok),
116            ColumnType::MYSQL_TYPE_MEDIUM_BLOB => MediumBlob(null_ok),
117            ColumnType::MYSQL_TYPE_LONG_BLOB => LongBlob(null_ok),
118            ColumnType::MYSQL_TYPE_JSON => Json(null_ok),
119            ColumnType::MYSQL_TYPE_VARCHAR => VarChar(null_ok),
120            ColumnType::MYSQL_TYPE_BIT => Bit(null_ok),
121            _ => unimplemented!("{}", format!("{:?}", ty)),
122        }
123    }
