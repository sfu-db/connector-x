1// Each variant in DataType represents a type that connectorx currently
2// supports to read from a data source and write into a destination.
3// When adding a new supported type T and associate it to the native representation N, please do
4// 1. Add a T variant to DataType.
5// 2. Add `DataType::T => N` to the macro impl_typesystem!.
6// 3. Add `DataType::T => N` to the macro impl_transmit!.
7//
8
9use chrono::{DateTime, Utc};
10/// This is a dummy type system used in this library.
11/// For all the sources, their output values must be one of the types defined by DummyTypeSystem.
12/// For all the destinations, they must support writing any value whose type is defined by DummyTypeSystem.
13#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
14pub enum CSVTypeSystem {
15    F64(bool),
16    I64(bool),
17    Bool(bool),
18    String(bool),
19    DateTime(bool),
20}
21
22impl_typesystem! {
23    system = CSVTypeSystem,
24    mappings = {
25        { F64 => f64 }
26        { I64 => i64 }
27        { Bool => bool }
28        { String => String }
29        { DateTime => DateTime<Utc> }
30    }
31}
