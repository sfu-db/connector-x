1use chrono::{DateTime, NaiveDateTime, Utc};
2use r2d2_oracle::oracle::sql_type::OracleType;
3use rust_decimal::Decimal;
4
5#[derive(Copy, Clone, Debug)]
6pub enum OracleTypeSystem {
7    NumInt(bool),
8    NumDecimal(bool),
9    Float(bool),
10    NumFloat(bool),
11    BinaryFloat(bool),
12    BinaryDouble(bool),
13    Blob(bool),
14    Clob(bool),
15    VarChar(bool),
16    Char(bool),
17    NVarChar(bool),
18    NChar(bool),
19    Date(bool),
20    Timestamp(bool),
21    TimestampTz(bool),
22    TimestampNano(bool),
23    TimestampTzNano(bool),
24}
25
26impl_typesystem! {
27    system = OracleTypeSystem,
28    mappings = {
29        { NumInt => i64 }
30        { NumDecimal => Decimal }
31        { Float | NumFloat | BinaryFloat | BinaryDouble => f64 }
32        { Blob => Vec<u8>}
33        { Clob | VarChar | Char | NVarChar | NChar => String }
34        { Date | Timestamp | TimestampNano => NaiveDateTime }
35        { TimestampTz | TimestampTzNano => DateTime<Utc> }
36    }
37}
38
39impl<'a> From<&'a OracleType> for OracleTypeSystem {
40    fn from(ty: &'a OracleType) -> OracleTypeSystem {
41        use OracleTypeSystem::*;
42        match ty {
43            OracleType::Number(0, 0) => NumFloat(true),
44            OracleType::Number(_, 0) => NumInt(true),
45            OracleType::Number(_, _) => NumDecimal(true),
46            OracleType::Float(_) => Float(true),
47            OracleType::BinaryFloat => BinaryFloat(true),
48            OracleType::BinaryDouble => BinaryDouble(true),
49            OracleType::BLOB => Blob(true),
50            OracleType::CLOB => Clob(true),
51            OracleType::Char(_) | OracleType::Long => Char(true),
52            OracleType::NChar(_) => NChar(true),
53            OracleType::Varchar2(_) => VarChar(true),
54            OracleType::NVarchar2(_) => NVarChar(true),
55            OracleType::Date => Date(true),
56            OracleType::Timestamp(7) | OracleType::Timestamp(8) | OracleType::Timestamp(9) => {
57                TimestampNano(true)
58            }
59            OracleType::Timestamp(_) => Timestamp(true),
60            OracleType::TimestampTZ(7)
61            | OracleType::TimestampTZ(8)
62            | OracleType::TimestampTZ(9) => TimestampTzNano(true),
63            OracleType::TimestampTZ(_) => TimestampTz(true),
64            _ => unimplemented!("{}", format!("Type {:?} not implemented for oracle!", ty)),
65        }
66    }
67}
