43impl_typesystem! {

74impl<'a> From<&'a Type> for PostgresTypeSystem {
75    fn from(ty: &'a Type) -> PostgresTypeSystem {
76        use PostgresTypeSystem::*;
77        match ty.name() {
78            "int2" => Int2(true),
79            "int4" => Int4(true),
80            "int8" => Int8(true),
81            "float4" => Float4(true),
82            "float8" => Float8(true),
83            "numeric" => Numeric(true),
84            "_bool" => BoolArray(true),
85            "_int2" => Int2Array(true),
86            "_int4" => Int4Array(true),
87            "_int8" => Int8Array(true),
88            "_float4" => Float4Array(true),
89            "_float8" => Float8Array(true),
90            "_numeric" => NumericArray(true),
91            "_varchar" => VarcharArray(true),
92            "_text" => TextArray(true),
93            "bool" => Bool(true),
94            "char" => Char(true),
95            "text" | "citext" | "ltree" | "lquery" | "ltxtquery" | "name" => Text(true),
96            "bpchar" => BpChar(true),
97            "varchar" => VarChar(true),
98            "bytea" => ByteA(true),
99            "time" => Time(true),
100            "timestamp" => Timestamp(true),
101            "timestamptz" => TimestampTz(true),
102            "date" => Date(true),
103            "uuid" => UUID(true),
104            "json" => JSON(true),
105            "jsonb" => JSONB(true),
106            "hstore" => HSTORE(true),
107            _ => match ty.kind() {
108                postgres::types::Kind::Enum(_) => Enum(true),
109                _ => unimplemented!("{}", ty.name()),
110            },
111        }
112    }
113}

115pub struct PostgresTypePairs<'a>(pub &'a Type, pub &'a PostgresTypeSystem);
116
117// Link (postgres::Type, connectorx::PostgresTypes) back to the one defiend by the postgres crate.
118impl From<PostgresTypePairs<'_>> for Type {
119    fn from(ty: PostgresTypePairs) -> Type {
120        use PostgresTypeSystem::*;
121        match ty.1 {
122            Enum(_) => Type::TEXT,
123            HSTORE(_) => Type::TEXT, // hstore is not supported in binary protocol (since no corresponding inner TYPE)
124            _ => ty.0.clone(),
125        }
126    }
127}
