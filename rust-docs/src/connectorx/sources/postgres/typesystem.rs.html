89impl<'a> From<&'a Type> for PostgresTypeSystem {
90    fn from(ty: &'a Type) -> PostgresTypeSystem {
91        use PostgresTypeSystem::*;
92        match ty.name() {
93            "int2" => Int2(true),
94            "int4" => Int4(true),
95            "int8" => Int8(true),
96            "oid" => UInt4(true),
97            "float4" => Float4(true),
98            "float8" => Float8(true),
99            "numeric" => Numeric(true),
100            "_bool" => BoolArray(true),
101            "_int2" => Int2Array(true),
102            "_int4" => Int4Array(true),
103            "_int8" => Int8Array(true),
104            "_float4" => Float4Array(true),
105            "_float8" => Float8Array(true),
106            "_numeric" => NumericArray(true),
107            "_varchar" => VarcharArray(true),
108            "_text" => TextArray(true),
109            "bool" => Bool(true),
110            "char" => Char(true),
111            "text" | "citext" | "ltree" | "lquery" | "ltxtquery" | "name" => Text(true),
112            "bpchar" => BpChar(true),
113            "varchar" => VarChar(true),
114            "bytea" => ByteA(true),
115            "time" => Time(true),
116            "timestamp" => Timestamp(true),
117            "timestamptz" => TimestampTz(true),
118            "date" => Date(true),
119            "uuid" => UUID(true),
120            "json" => JSON(true),
121            "jsonb" => JSONB(true),
122            "hstore" => HSTORE(true),
123            "inet" => Inet(true),
124            "vector" => Vector(true),
125            "halfvec" => HalfVec(true),
126            "bit" => Bit(true),
127            "sparsevec" => SparseVec(true),
128            _ => match ty.kind() {
129                postgres::types::Kind::Enum(_) => Enum(true),
130                _ => unimplemented!("{}", ty.name()),
131            },
132        }
133    }
134}

136pub struct PostgresTypePairs<'a>(pub &'a Type, pub &'a PostgresTypeSystem);
137
138// Link (postgres::Type, connectorx::PostgresTypes) back to the one defined by the postgres crate.
139impl From<PostgresTypePairs<'_>> for Type {
140    fn from(ty: PostgresTypePairs) -> Type {
141        use PostgresTypeSystem::*;
142        match ty.1 {
143            Enum(_) => Type::TEXT,
144            HSTORE(_) => Type::TEXT, // hstore is not supported in binary protocol (since no corresponding inner TYPE)
145            _ => ty.0.clone(),
146        }
147    }
148}
