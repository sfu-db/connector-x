22impl_typesystem! {
23    system = BigQueryTypeSystem,
24    mappings = {
25        { Bool | Boolean => bool }
26        { Int64 | Integer  =>  i64 }
27        { Float64 | Float | Numeric | Bignumeric  =>  f64 }
28        { String | Bytes  =>  String }
29        { Date => NaiveDate }
30        { Datetime => NaiveDateTime }
31        { Time => NaiveTime }
32        { Timestamp => DateTime<Utc> }
33    }
34}

36impl<'a> From<&'a FieldType> for BigQueryTypeSystem {
37    fn from(ty: &'a FieldType) -> BigQueryTypeSystem {
38        use BigQueryTypeSystem::*;
39        match ty {
40            FieldType::Bool => Bool(true),
41            FieldType::Boolean => Boolean(true),
42            FieldType::Int64 => Int64(true),
43            FieldType::Integer => Integer(true),
44            FieldType::Float => Float(true),
45            FieldType::Float64 => Float64(true),
46            FieldType::Numeric => Numeric(true),
47            FieldType::Bignumeric => Bignumeric(true),
48            FieldType::String => String(true),
49            FieldType::Bytes => Bytes(true),
50            FieldType::Date => Date(true),
51            FieldType::Datetime => Datetime(true),
52            FieldType::Time => Time(true),
53            FieldType::Timestamp => Timestamp(true),
54            _ => unimplemented!("{}", format!("{:?}", ty)),
55        }
56    }
57}
