1use super::errors::SQLiteSourceError;
2use chrono::{NaiveDate, NaiveDateTime, NaiveTime};
3use fehler::{throw, throws};
4use rusqlite::types::Type;
5use std::convert::TryFrom;
6
7#[derive(Copy, Clone, Debug, PartialEq)]
8pub enum SQLiteTypeSystem {
9    Bool(bool),
10    Int8(bool),
11    Int4(bool),
12    Int2(bool),
13    Real(bool),
14    Text(bool),
15    Date(bool),
16    Time(bool),
17    Timestamp(bool),
18    Blob(bool),
19}
20
21impl_typesystem! {
22    system = SQLiteTypeSystem,
23    mappings = {
24        { Bool => bool }
25        { Int8 => i64 }
26        { Int4 => i32 }
27        { Int2 => i16 }
28        { Real => f64 }
29        { Text => Box<str> }
30        { Date => NaiveDate}
31        { Time => NaiveTime}
32        { Timestamp => NaiveDateTime}
33        { Blob => Vec<u8>}
34    }
35}
36
37impl TryFrom<Type> for SQLiteTypeSystem {
38    type Error = SQLiteSourceError;
39
40    #[throws(SQLiteSourceError)]
41    fn try_from(ty: Type) -> Self {
42        use SQLiteTypeSystem::*;
43        match ty {
44            Type::Integer => Int8(true),
45            Type::Real => Real(true),
46            Type::Text => Text(true),
47            Type::Blob => Blob(true),
48            Type::Null => throw!(SQLiteSourceError::InferTypeFromNull),
49        }
50    }
51}
52
53impl TryFrom<(Option<&str>, Type)> for SQLiteTypeSystem {
54    type Error = SQLiteSourceError;
55
56    #[throws(SQLiteSourceError)]
57    fn try_from(types: (Option<&str>, Type)) -> Self {
58        use SQLiteTypeSystem::*;
59        match types {
60            // derive from column's declare type, some rules refer to:
61            // https://www.sqlite.org/datatype3.html#affname
62            (Some(decl_type), ty) => {
63                let decl_type = decl_type.to_lowercase();
64                match decl_type.as_str() {
65                    "int4" => Int4(true),
66                    "int2" => Int2(true),
67                    "boolean" | "bool" => Bool(true),
68                    "date" => Date(true),
69                    "time" => Time(true),
70                    "datetime" | "timestamp" => Timestamp(true),
71                    _ if decl_type.contains("int") => Int8(true),
72                    _ if decl_type.contains("char")
73                        || decl_type.contains("clob")
74                        || decl_type.contains("text") =>
75                    {
76                        Text(true)
77                    }
78                    _ if decl_type.contains("real")
79                        || decl_type.contains("floa")
80                        || decl_type.contains("doub") =>
81                    {
82                        Real(true)
83                    }
84                    _ if decl_type.contains("blob") => Blob(true),
85                    _ => SQLiteTypeSystem::try_from(ty)?,
86                }
87            }
88            // derive from value type directly if no declare type available
89            (None, ty) => SQLiteTypeSystem::try_from(ty)?,
90        }
91    }
92}
