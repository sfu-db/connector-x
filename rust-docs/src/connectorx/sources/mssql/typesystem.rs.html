40impl_typesystem! {
41    system = MsSQLTypeSystem,
42    mappings = {
43        { Tinyint  => u8 }
44        { Smallint => i16 }
45        { Int => i32 }
46        { Bigint => i64 }
47        { Intn => IntN }
48        { Float24 | SmallMoney => f32 }
49        { Float53 | Money => f64 }
50        { Floatn => FloatN }
51        { Bit => bool }
52        { Nvarchar | Varchar | Nchar | Char | Text | Ntext => &'r str }
53        { Binary | Varbinary | Image => &'r [u8] }
54        { Uniqueidentifier => Uuid }
55        { Numeric | Decimal => Decimal }
56        { Datetime | Datetime2 | Smalldatetime => NaiveDateTime }
57        { Date => NaiveDate }
58        { Time => NaiveTime }
59        { Datetimeoffset => DateTime<Utc> }

63impl<'a> From<&'a ColumnType> for MsSQLTypeSystem {

78            ColumnType::NVarchar => Nvarchar(true),

80            ColumnType::NChar => Nchar(true),
81            ColumnType::BigChar => Char(true),
82            ColumnType::NText => Ntext(true),
83            ColumnType::Text => Text(true),
84            ColumnType::BigBinary => Binary(true),
85            ColumnType::BigVarBin => Varbinary(true),

104pub struct IntN(pub i64);
105impl<'a> FromSql<'a> for IntN {
106    fn from_sql(value: &'a ColumnData<'static>) -> Result<Option<Self>, tiberius::error::Error> {
107        match value {
108            ColumnData::U8(None)
109            | ColumnData::I16(None)
110            | ColumnData::I32(None)
111            | ColumnData::I64(None) => Ok(None),
112            ColumnData::U8(Some(d)) => Ok(Some(IntN(*d as i64))),
113            ColumnData::I16(Some(d)) => Ok(Some(IntN(*d as i64))),
114            ColumnData::I32(Some(d)) => Ok(Some(IntN(*d as i64))),
115            ColumnData::I64(Some(d)) => Ok(Some(IntN(*d))),
116            v => Err(tiberius::error::Error::Conversion(
117                format!("cannot interpret {:?} as a intn value", v).into(),
118            )),
119        }
120    }
121}

123pub struct FloatN(pub f64);

125    fn from_sql(value: &'a ColumnData<'static>) -> Result<Option<Self>, tiberius::error::Error> {
126        match value {
127            ColumnData::F32(None) | ColumnData::F64(None) => Ok(None),
128            ColumnData::F32(Some(d)) => Ok(Some(FloatN(*d as f64))),
129            ColumnData::F64(Some(d)) => Ok(Some(FloatN(*d))),
130            v => Err(tiberius::error::Error::Conversion(
131                format!("cannot interpret {:?} as a floatn value", v).into(),
132            )),
133        }
134    }
