149impl ClickHouseTypeSystem {
150    pub fn from_type_str_with_metadata(type_str: &str) -> (Self, TypeMetadata) {
151        use ClickHouseTypeSystem::*;
152
153        let type_str = type_str.trim();
154        let mut metadata = TypeMetadata::default();
155
156        let (type_str, nullable) = Self::unwrap_nullable(type_str);
157
158        if let Some(inner) = Self::unwrap_wrapper(type_str, "LowCardinality") {
159            return Self::from_type_str_with_metadata(inner);
160        }
161
162        if let Some(inner) = Self::unwrap_wrapper(type_str, "Array") {
163            if let Some(array_type) = Self::parse_array_type(inner, nullable) {
164                return (array_type, metadata);
165            }
166            return (String(nullable), metadata);
167        }
168
169        if let Some(params) = Self::unwrap_wrapper(type_str, "Enum8") {
170            metadata.named_values = Self::parse_enum_definition(params);
171            return (Enum8(nullable), metadata);
172        }
173        if let Some(params) = Self::unwrap_wrapper(type_str, "Enum16") {
174            metadata.named_values = Self::parse_enum_definition(params);
175            return (Enum16(nullable), metadata);
176        }
177
178        let (base_type, params) = Self::split_type_params(type_str);
179
180        let ts = match base_type {
181            "Int8" => Int8(nullable),
182            "Int16" => Int16(nullable),
183            "Int32" => Int32(nullable),
184            "Int64" => Int64(nullable),
185            "UInt8" => UInt8(nullable),
186            "UInt16" => UInt16(nullable),
187            "UInt32" => UInt32(nullable),
188            "UInt64" => UInt64(nullable),
189            "Float32" => Float32(nullable),
190            "Float64" => Float64(nullable),
191            "Decimal" => {
192                let (precision, scale) = Self::parse_decimal_precision_scale(params);
193                metadata.precision = precision;
194                metadata.scale = scale;
195                Decimal(nullable)
196            }
197            "String" => String(nullable),
198            "FixedString" => {
199                metadata.length = Self::parse_length(params);
200                FixedString(nullable)
201            }
202            "Date" => Date(nullable),
203            "Date32" => Date32(nullable),
204            "Time" => Time(nullable),
205            "Time64" => {
206                metadata.precision = Self::parse_time64_precision(params);
207                Time64(nullable)
208            }
209            "DateTime" => {
210                metadata.timezone = Self::parse_datetime_params(params).1;
211                DateTime(nullable)
212            }
213            "DateTime64" => {
214                let (precision, tz) = Self::parse_datetime_params(params);
215                metadata.precision = precision.unwrap_or(3);
216                metadata.timezone = tz;
217                DateTime64(nullable)
218            }
219            "UUID" => UUID(nullable),
220            "Bool" => Bool(nullable),
221            "IPv4" => IPv4(nullable),
222            "IPv6" => IPv6(nullable),
223            "Enum8" => {
224                if let Some(params) = params {
225                    metadata.named_values = Self::parse_enum_definition(params);
226                }
227                Enum8(nullable)
228            }
229            "Enum16" => {
230                if let Some(params) = params {
231                    metadata.named_values = Self::parse_enum_definition(params);
232                }
233                Enum16(nullable)
234            }
235            _ => String(nullable),
236        };
237
238        (ts, metadata)
239    }

241    pub fn from_type_str(type_str: &str) -> Self {
242        Self::from_type_str_with_metadata(type_str).0
243    }

245    fn unwrap_nullable(s: &str) -> (&str, bool) {
246        if let Some(inner) = Self::unwrap_wrapper(s, "Nullable") {
247            (inner, true)
248        } else {
249            (s, false)
250        }
251    }
252
253    /// Unwrap "Wrapper(inner)" -> Some(inner), or None if not matching
254    fn unwrap_wrapper<'a>(s: &'a str, wrapper: &str) -> Option<&'a str> {
255        let prefix = format!("{}(", wrapper);
256        if s.starts_with(&prefix) && s.ends_with(')') {
257            Some(&s[prefix.len()..s.len() - 1])
258        } else {
259            None
260        }
261    }
262
263    fn split_type_params(s: &str) -> (&str, Option<&str>) {
264        if let Some(idx) = s.find('(') {
265            if s.ends_with(')') {
266                let base = &s[..idx];
267                let params_str = &s[idx + 1..s.len() - 1];
268                return (base, Some(params_str));
269            }
270        }
271        (s, None)
272    }
273
274    /// Parse FixedString(N)
275    fn parse_length(params: Option<&str>) -> usize {
276        let params = params.and_then(|p| p.split(',').map(|i| i.trim()).collect::<Vec<_>>().into());
277        params
278            .and_then(|p| p.get(0).and_then(|s| s.parse::<usize>().ok()))
279            .unwrap_or(1)
280    }
281
282    /// Parse DateTime(precision, 'Timezone') or DateTime('Timezone')
283    fn parse_datetime_params(params: Option<&str>) -> (Option<u8>, Option<Tz>) {
284        let params = params.and_then(|p| p.split(',').map(|i| i.trim()).collect::<Vec<_>>().into());
285        match params {
286            None => (None, None),
287            Some(p) => {
288                if let Some(precision) = p.get(0).and_then(|s| s.parse::<u8>().ok()) {
289                    let timezone = p.get(1).and_then(|s| Self::parse_timezone(s));
290                    (Some(precision), timezone)
291                } else {
292                    (None, p.get(0).and_then(|s| Self::parse_timezone(s)))
293                }
294            }
295        }
296    }
297
298    /// Parse Time64(precision)
299    fn parse_time64_precision(params: Option<&str>) -> u8 {
300        let params = params.and_then(|p| p.split(',').map(|i| i.trim()).collect::<Vec<_>>().into());
301        params
302            .and_then(|p| p.get(0).and_then(|s| s.parse::<u8>().ok()))
303            .unwrap_or(3)
304    }
305
306    /// Parse Decimal(precision, scale)
307    fn parse_decimal_precision_scale(params: Option<&str>) -> (u8, u8) {
308        let params = params.and_then(|p| p.split(',').map(|i| i.trim()).collect::<Vec<_>>().into());
309        params
310            .and_then(|p| {
311                let precision = p.get(0).and_then(|s| s.parse::<u8>().ok());
312                let scale = p.get(1).and_then(|s| s.parse::<u8>().ok());
313                Some((precision.unwrap_or(0), scale.unwrap_or(0)))
314            })
315            .unwrap_or((0, 0))
316    }
317
318    fn parse_timezone(s: &str) -> Option<Tz> {
319        s.trim_matches('\'').parse::<Tz>().ok()
320    }
321
322    /// Parse Array(InnerType) and return the corresponding ArrayXxx variant
323    fn parse_array_type(inner_type: &str, nullable: bool) -> Option<Self> {
324        use ClickHouseTypeSystem::*;
325
326        let inner = inner_type.trim();
327
328        let inner = if let Some(unwrapped) = Self::unwrap_wrapper(inner, "Nullable") {
329            unwrapped
330        } else {
331            inner
332        };
333
334        let (base, _) = Self::split_type_params(inner);
335
336        match base {
337            "Bool" => Some(ArrayBool(nullable)),
338            "String" => Some(ArrayString(nullable)),
339            "Int8" => Some(ArrayInt8(nullable)),
340            "Int16" => Some(ArrayInt16(nullable)),
341            "Int32" => Some(ArrayInt32(nullable)),
342            "Int64" => Some(ArrayInt64(nullable)),
343            "UInt8" => Some(ArrayUInt8(nullable)),
344            "UInt16" => Some(ArrayUInt16(nullable)),
345            "UInt32" => Some(ArrayUInt32(nullable)),
346            "UInt64" => Some(ArrayUInt64(nullable)),
347            "Float32" => Some(ArrayFloat32(nullable)),
348            "Float64" => Some(ArrayFloat64(nullable)),
349            "Decimal" => Some(ArrayDecimal(nullable)),
350            _ => None,
351        }
352    }
353
354    /// Parse Enum8/Enum16 definitions like "'a' = 1, 'b' = 2, 'c' = 3"
355    fn parse_enum_definition(params: &str) -> Option<FxHashMap<i16, String>> {
356        let re = Regex::new(r"'((?:\\'|[^'])*)'\s*=\s*(-?\d+)").unwrap();
357
358        re.captures_iter(params)
359            .map(|cap| {
360                let key = cap[1].replace("\\'", "'");
361                let value = cap[2].parse().unwrap();
362                (value, key)
363            })
364            .collect::<FxHashMap<_, _>>()
365            .into()
366    }

373            | ClickHouseTypeSystem::Int64(nullable)
374            | ClickHouseTypeSystem::UInt8(nullable)
375            | ClickHouseTypeSystem::UInt16(nullable)
376            | ClickHouseTypeSystem::UInt32(nullable)
377            | ClickHouseTypeSystem::UInt64(nullable)
378            | ClickHouseTypeSystem::Float32(nullable)
379            | ClickHouseTypeSystem::Float64(nullable)
380            | ClickHouseTypeSystem::Decimal(nullable)
381            | ClickHouseTypeSystem::String(nullable)
382            | ClickHouseTypeSystem::FixedString(nullable)
383            | ClickHouseTypeSystem::Date(nullable)
384            | ClickHouseTypeSystem::Date32(nullable)
385            | ClickHouseTypeSystem::Time(nullable)
386            | ClickHouseTypeSystem::Time64(nullable)
387            | ClickHouseTypeSystem::DateTime(nullable)
388            | ClickHouseTypeSystem::DateTime64(nullable)
389            | ClickHouseTypeSystem::Enum8(nullable)

392            | ClickHouseTypeSystem::IPv4(nullable)
