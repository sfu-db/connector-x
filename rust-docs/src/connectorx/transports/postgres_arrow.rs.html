24#[derive(Error, Debug)]
25pub enum PostgresArrowTransportError {
26    #[error(transparent)]
27    Source(#[from] PostgresSourceError),
28
29    #[error(transparent)]
30    Destination(#[from] ArrowDestinationError),
31
32    #[error(transparent)]
33    ConnectorX(#[from] crate::errors::ConnectorXError),
34}

36/// Convert Postgres data types to Arrow data types.
37pub struct PostgresArrowTransport<P, C>(PhantomData<P>, PhantomData<C>);
38
39macro_rules! impl_postgres_transport {
40    ($proto:ty, $tls:ty) => {
41        impl_transport!(
42            name = PostgresArrowTransport<$proto, $tls>,
43            error = PostgresArrowTransportError,
44            systems = PostgresTypeSystem => ArrowTypeSystem,
45            route = PostgresSource<$proto, $tls> => ArrowDestination,
46            mappings = {
47                { Float4[f32]                        => Float32[f32]                           | conversion auto   }
48                { Float8[f64]                        => Float64[f64]                           | conversion auto   }
49                { Numeric[Decimal]                   => Decimal[Decimal]                       | conversion auto   }
50                { Int2[i16]                          => Int16[i16]                             | conversion auto   }
51                { Int4[i32]                          => Int32[i32]                             | conversion auto   }
52                { Int8[i64]                          => Int64[i64]                             | conversion auto   }
53                { Bool[bool]                         => Boolean[bool]                          | conversion auto   }
54                { Text[&'r str]                      => LargeUtf8[String]                      | conversion owned  }
55                { BpChar[&'r str]                    => LargeUtf8[String]                      | conversion none   }
56                { VarChar[&'r str]                   => LargeUtf8[String]                      | conversion none   }
57                { Name[&'r str]                      => LargeUtf8[String]                      | conversion none   }
58                { Enum[&'r str]                      => LargeUtf8[String]                      | conversion none   }
59                { Timestamp[NaiveDateTime]           => Date64Micro[NaiveDateTimeWrapperMicro] | conversion option }
60                { Date[NaiveDate]                    => Date32[NaiveDate]                      | conversion auto   }
61                { Time[NaiveTime]                    => Time64Micro[NaiveTimeWrapperMicro]     | conversion option }
62                { TimestampTz[DateTime<Utc>]         => DateTimeTzMicro[DateTimeWrapperMicro]  | conversion option }
63                { UUID[Uuid]                         => LargeUtf8[String]                      | conversion option }
64                { Char[&'r str]                      => LargeUtf8[String]                      | conversion none   }
65                { ByteA[Vec<u8>]                     => LargeBinary[Vec<u8>]                   | conversion auto   }
66                { JSON[Value]                        => LargeUtf8[String]                      | conversion option }
67                { JSONB[Value]                       => LargeUtf8[String]                      | conversion none   }
68                { BoolArray[Vec<Option<bool>>]       => BoolArray[Vec<Option<bool>>]           | conversion auto   }
69                { VarcharArray[Vec<Option<String>>]  => Utf8Array[Vec<Option<String>>]         | conversion auto   }
70                { TextArray[Vec<Option<String>>]     => Utf8Array[Vec<Option<String>>]         | conversion none   }
71                { Int2Array[Vec<Option<i16>>]        => Int16Array[Vec<Option<i16>>]           | conversion auto   }
72                { Int4Array[Vec<Option<i32>>]        => Int32Array[Vec<Option<i32>>]           | conversion auto   }
73                { Int8Array[Vec<Option<i64>>]        => Int64Array[Vec<Option<i64>>]           | conversion auto   }
74                { Float4Array[Vec<Option<f32>>]      => Float32Array[Vec<Option<f32>>]         | conversion auto   }
75                { Float8Array[Vec<Option<f64>>]      => Float64Array[Vec<Option<f64>>]         | conversion auto   }
76                { NumericArray[Vec<Option<Decimal>>] => DecimalArray[Vec<Option<Decimal>>]     | conversion auto   }
77            }
78        );
79    }
80}
