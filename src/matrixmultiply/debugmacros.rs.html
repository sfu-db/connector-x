// for debugging -- like println during debugging
macro_rules! dprint {
    ($($t:tt)*) => {
        debug!(println!($($t)*))
    }
}

#[cfg(feature = "use_debug")]
macro_rules! debug {
    ($e:expr) => {
        $e;
    }
}

#[cfg(not(feature = "use_debug"))]
macro_rules! debug {
    ($e:expr) => {
    }
}
